    ylabel={Betrag [dB]},
    grid=both,
    major grid style={line width=.2pt,draw=gray!50},
    minor grid style={line width=.1pt,draw=gray!30},
    axis lines=left,
    xtick={1e-3,0.01,0.1,1,10,100,1000},
    ytick={-60,-40,-20,0,20,40,60}
]
